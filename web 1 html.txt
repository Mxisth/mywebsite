function goto(screen) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen-' + screen).classList.add('active');
}

function switchTab(el, tabId) {
  const parent = el.closest('.nav-tabs');
  parent.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');

  const mainArea = el.closest('.main');
  mainArea.querySelectorAll('[id^="tab-"], [id^="atab-"]').forEach(t => t.style.display = 'none');
  document.getElementById(tabId).style.display = '';
}

function doLogin() {
  const email = document.getElementById('login-email').value.trim();
  const pass  = document.getElementById('login-pass').value;
  const errEl = document.getElementById('login-error');

  const user = DB.users.find(u => u.email === email && u.passwordHash === pass);
  if (!user) {
    // Log failed attempt
    DB.logs.unshift({
      id: 'log_' + Date.now(),
      userId: null,
      eventType: 'failed_login',
      ip: randomIP(),
      device: getDevice(),
      os: getOS(),
      browser: getBrowser(),
      country: 'TH',
      ts: new Date()
    });
    errEl.classList.add('show');
    return;
  }
  errEl.classList.remove('show');

  // Log login
  DB.logs.unshift({
    id: 'log_' + Date.now(),
    userId: user.id,
    eventType: 'login',
    ip: randomIP(),
    device: getDevice(),
    os: getOS(),
    browser: getBrowser(),
    country: 'TH',
    ts: new Date()
  });

  user.lastLoginAt = new Date();
  currentUser = user;
  showToast('✅', `Welcome back, ${user.name.split(' ')[0]}!`);

  if (user.role === 'admin') {
    renderAdminDash();
    goto('admin');
  } else {
    renderUserDash();
    goto('dashboard');
  }
}

function doSignup() {
  const name    = document.getElementById('signup-name').value.trim();
  const email   = document.getElementById('signup-email').value.trim();
  const pass    = document.getElementById('signup-pass').value;
  const confirm = document.getElementById('signup-confirm').value;
  const errEl   = document.getElementById('signup-error');
  const okEl    = document.getElementById('signup-success');

  errEl.classList.remove('show');
  okEl.classList.remove('show');

  if (!name || !email || !pass) { errEl.textContent = '❌ Please fill in all fields'; errEl.classList.add('show'); return; }
  if (pass.length < 6)          { errEl.textContent = '❌ Password must be at least 6 characters'; errEl.classList.add('show'); return; }
  if (pass !== confirm)         { errEl.textContent = '❌ Passwords do not match'; errEl.classList.add('show'); return; }
  if (DB.users.find(u => u.email === email)) { errEl.textContent = '❌ Email already registered'; errEl.classList.add('show'); return; }

  const newUser = {
    id: 'usr_' + Date.now(),
    name, email,
    passwordHash: pass,
    role: 'user',
    isActive: true,
    createdAt: new Date(),
    lastLoginAt: null
  };
  DB.users.push(newUser);

  DB.logs.unshift({
    id: 'log_' + Date.now(),
    userId: newUser.id,
    eventType: 'signup',
    ip: randomIP(),
    device: getDevice(),
    os: getOS(),
    browser: getBrowser(),
    country: 'TH',
    ts: new Date()
  });

  okEl.textContent = '✅ Account created! You can now sign in.';
  okEl.classList.add('show');
  showToast('🎉', 'Account created successfully!');

  setTimeout(() => {
    document.getElementById('login-email').value = email;
    document.getElementById('login-pass').value = pass;
    goto('login');
  }, 1500);
}

  currentUser = null;
  setTimeout(() => goto('login'), 300);
}

function renderUsersTable() {

    const matchRole   = !roleF  || u.role === roleF;
    return matchSearch && matchRole;
  });

  document.getElementById('users-tbody').innerHTML = users.map(u => {
    const lastLog = DB.logs.find(l => l.userId === u.id && l.eventType === 'login');
    return `
    <tr>
      <td>
        <div style="display:flex;align-items:center;gap:10px">

function deleteUser(id) {

function logEntryHTML(l, showUser = false) {

        <span>🌐 ${l.ip}</span>
        <span>📱 ${l.device}</span>
        <span>🖥 ${l.os}</span>
        <span>🌍 ${l.country}</span>
      </div>
    </div>
    <div class="log-time">${fmtDate(l.ts)}</div>
  </div>`;
}

function eventStyle(type) {
  const map = {
    login:        { icon: '✅', color: 'var(--green)' },
    logout:       { icon: '🔓', color: 'var(--accent)' },
    signup:       { icon: '🎉', color: 'var(--amber)' },
    failed_login: { icon: '❌', color: 'var(--red)' },
  };
  return map[type] || { icon: '•', color: 'var(--text2)' };
}

function emptyState(msg) {

function fmtDate(d, short = false) {
  if (!d) return '—';
  if (short) return d.toLocaleDateString('en-GB', { day:'2-digit', month:'short', year:'numeric' });
  return d.toLocaleString('en-GB', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });
}

function getGreeting() {
  const h = new Date().getHours();
  if (h < 12) return 'morning';
  if (h < 17) return 'afternoon';
  return 'evening';
}

function randomIP() {
  const pools = ['171.96','203.150','49.229','110.170','1.46'];
  const pool = pools[Math.floor(Math.random()*pools.length)];
  return `${pool}.${Math.floor(Math.random()*255)}.${Math.floor(Math.random()*255)}`;
}

function getDevice() {
  const ua = navigator.userAgent;
  if (/Mobile|Android|iPhone/i.test(ua)) return 'Mobile';
  if (/iPad|Tablet/i.test(ua)) return 'Tablet';
  return 'Desktop';
}

function getOS() {
  const ua = navigator.userAgent;
  if (/Windows NT 10/i.test(ua)) return 'Windows 11';
  if (/Windows/i.test(ua)) return 'Windows';
  if (/Mac OS X/i.test(ua)) return 'macOS';
  if (/iPhone|iPad/i.test(ua)) return 'iOS';
  if (/Android/i.test(ua)) return 'Android';
  if (/Linux/i.test(ua)) return 'Linux';
  return 'Unknown OS';
}

function getBrowser() {
  const ua = navigator.userAgent;
  if (/Edg/i.test(ua)) return 'Edge';
  if (/Chrome/i.test(ua)) return 'Chrome';
  if (/Firefox/i.test(ua)) return 'Firefox';
  if (/Safari/i.test(ua)) return 'Safari';
  return 'Browser';
}
